\begin{frame}
\frametitle{Developing mechanical models through the principle of virtual powers}
\centering
\begin{minipage}[c]{0.47\textwidth}
\textbf{Description of a mechanical system}
\\[0.5em]
The systems are the elements $z$ of a separable and reflexive Banach space $Z$.
\\[0.5em]
Generalized forces are the elements $\mathcal{F}$ of the dual space $Z^*$.
\\[1em]
\textbf{Equilibrium}
\\[0.5em]
Equilibrium is the balance
\[
-\mathcal{I}(t)+\mathcal{X}(t)=0\quad\text{for a.e.\ $t\in(0,T)$}
\]
between internal forces $\mathcal{I}\colon(0,T)\to Z^*$ and external forces $\mathcal{X}\colon(0,T)\to Z^*$.
\end{minipage}
\hspace{0.02\textwidth}
\begin{minipage}[c]{0.47\textwidth}
\textbf{Constitutive relations}
\\[0.5em]
Let $\Psi\colon Z\to\mathbb{R}\cup\{0\}$ be the free energy and $\Phi\colon Z\to[0,\infty]$ the dissipation potential.
\\[0.5em]
Internal forces are related to the evolution $z\colon[0,T]\to Z$ with a constitutive relation
\[
\mathcal{I}(t)\in D\Psi(z(t))+\partial\Phi(\dot z(t))\quad\text{for a.e.\ $t\in(0,T)$}.
\]
\textsc{Definition.} $\xi\in\partial\Phi(v)$ if $\Phi(w)\ge\Phi(v)+\dual{\xi,w-v}$ for all $w\in Z$.
\\[0.5em]
\textsc{Proposition.} If $\Phi$ is convex and $\Phi(0)=0$ then $\dual{\xi,v}\ge0$ for all $\xi\in\partial\Phi(v)$.
\\[0.5em]
\textsc{Corollary.} $\frac{d}{dt}\Psi(z(t))\le\dual{\mathcal{I}(t),\dot z(t)}=\dual{\mathcal{X}(t),\dot z(t)}$.
\end{minipage}
\\[2em]
\begin{minipage}[c]{0.4\textwidth}
\textsc{Problem.} Given $z_0\in Z$ and $\mathcal{X}$ find $z$ such that $z(0)=z_0$ and equilibrium holds.
\end{minipage}
$\qquad\implies\qquad$
\begin{minipage}[c]{0.4\textwidth}
Doubly nonlinear equation
\[
D\Psi(z(t))+\partial\Phi(\dot z(t))\ni\mathcal{X}(t)
\]
for a.e.\ $t\in(0,T)$
\end{minipage}
\end{frame}

\begin{frame}
\frametitle{Treating constraints with the method of Lagrange multipliers}
\centering
\begin{minipage}[c]{0.47\textwidth}
\textbf{Constrained kinematics}
\\[0.5em]
Let $K\subset Z$ closed and convex and impose the constraint $z(t)\in K$ for all $t\in[0,T]$.
\\[1em]
\textbf{Reaction force}
\\[0.5em]
Add a reaction force $\mathcal{P}\colon(0,T)\to Z^*$ satisfying $\mathcal{P}(t)\in-N_K(z(t))=-\partial\iota_K(z(t))$ for a.e.\ $t\in(0,T)$.
\\[1em]
\textbf{Equilibrium}
\\[0.5em]
The equilibrium now reads
\[
-\mathcal{I}(t)+\mathcal{X}(t)+\mathcal{P}(t)=0\quad\text{for a.e.\ $t\in(0,T)$}.
\]
\end{minipage}
\hspace{0.02\textwidth}
\begin{minipage}[c]{0.47\textwidth}
\textbf{Constitutive relations}
\\[0.5em]
Let again $\Psi\colon Z\to\mathbb{R}\cup\{0\}$ be the free energy.
\\[0.5em]
Now the dissipation potential $\Phi\colon Z^*\times Z\to[0,\infty],\;(\mathcal{P},w)\mapsto\Phi(\mathcal{P},w)$ can depend on the reaction force.
\\[0.5em]
The constitutive relations become
\[
\mathcal{I}(t)\in D\Psi(z(t))+\partial_w\Phi(\mathcal{P}(t),\dot z(t))
\]
for a.e.\ $t\in(0,T)$.
\end{minipage}
\\[1em]
\begin{minipage}[c]{0.4\textwidth}
\textsc{Problem.} Given $z_0\in Z$ and $\mathcal{X}$ find $z$ and $\mathcal{P}$ such that $z(0)=z_0$, $z(t)\in K$ for all $t\in[0,T]$, $\mathcal{P}(t)\in-N_K(z(t))$ for a.e.\ $t\in(0,T)$ and equilibrium holds.
\end{minipage}
$\qquad\implies\qquad$
\begin{minipage}[c]{0.4\textwidth}
Constrained doubly nonlinear equation
\begin{gather*}
z(t)\in K\quad\text{for all $t\in[0,T]$} \\
\mathcal{P}(t)\in-N_K(z(t)) \\
D\Psi(z(t))+\partial\Phi(\mathcal{P}(t),\dot z(t))\ni\mathcal{X}(t)+\mathcal{P}(t)
\end{gather*}
for a.e.\ $t\in(0,T)$
\end{minipage}
\end{frame}

\begin{frame}
\frametitle{Classical approach}
\centering
\begin{minipage}[c]{0.6\textwidth}
\textbf{Kinematics}
\\[0.5em]
The space of configurations is $Z=\{(u,E_p)\}$ with the displacement $u\in H^1(\Omega,\mathbb{R}^d)$ and the plastic strain $E_p\in H^1(\Omega,\text{Sym})$. The elastic strain is $E_e=Eu-E_p$.
\\[0.5em]
The generic rate is $w=(v,V_p)$ and the elastic rate is $V_e=Ev-V_p$.
\\[1em]
\textbf{Statics}
\\[0.5ex]
The internal generalized force is
\[
\dual{\mathcal{I},w}=\int_\Omega\left[T_e:V_e+T_p:\grad V_p+K_p\threedots\grad V_p\right]dx
\]
with $T_e\colon\Omega\to\text{Sym}$ the elastic stress, $T_p\colon\Omega\to\text{Sym}$ the plastic force and $K_p\colon\Omega\to\text{Sym}\otimes\mathbb{R}$ the plastic stress. The external generalized force is
\[
\dual{\mathcal{X},w}=\int_\Omega f\cdot vdx+\int_{\partial\Omega}h\cdot vdx.
\]
\end{minipage}
\hspace{0.02\textwidth}
\begin{minipage}[c]{0.33\textwidth}
\begin{center}

\vspace{2em}
\textsc{Proposition.} The strong form of the equilibrium is
\[
\begin{gathered}
\diver T_e+f=0\\
T_e-T_p+\diver K_p=0
\end{gathered}
\quad\text{in $\Omega$}
\]
and $T_en=h$, $K_pn=0$ in $\partial\Omega$.

\begin{frame}
\frametitle{Classical approach}
\centering
\begin{minipage}[c]{0.45\textwidth}
\textbf{Constitutive relations}
\\[0.5ex]
Let the free energy
\[
\Psi(z)=\int_\Omega\left[\frac{1}{2}\mathfrak{C}E_e:E_e+\frac{1}{2}\mathfrak{b}|K_p|^2\right]dx
\]
that implies
\[
T_e=\mathfrak{C}E_e\quad\text{and}\quad K_p=\mathfrak{b}\grad E_p.
\]
\textbf{Yield condition and non-associated flow rule}
\\[0.5em]
Fix $f,g\colon\text{Sym}\to\mathbb{R}$ convex. Let $C=\{f\le0\}$ and $L=\{g\le0\}$.
\\[0.5em]
Impose that $T_p\in C$. The yield condition is $T_p\in\partial C$ and the flow rule is
\[
\dot E_p\in
\begin{cases}
\{0\} & \text{if $T_p\in\inter C$} \\
N_{L}(T_p) & \text{if $T_p\in\partial C$}.
\end{cases}
\]
\textsc{Proposition}. If $f=g$, is equivalent to $T_p\in\partial\sigma_C(\dot E_p)$ with $\sigma_C(\dot E_p)=\sup_{T_p\in C}T_p:\dot E_p$.
\end{minipage}
\hspace{0.02\textwidth}
\begin{minipage}[c]{0.45\textwidth}
\textsc{Example.} Drucker--Prager yield criterion
\begin{gather*}
f(T_p)=|\dev T_p|-\left(f_0-a\frac{\tr T_p}{d}\right) \\
g(T_p)=|\dev T_p|+b\frac{\tr T_p}{d}
\end{gather*}
with $0\le b\le a$.
\vspace{1em}
\begin{center}
\begin{tikzpicture}[scale=0.7]

\draw (0,0)--(0,2);

\node[anchor=west] at (2,0) {$\dfrac{\tr T_p}{d}$};

\draw[dashed] ({0.5},{-1.0/2.1333*(0.5-1.5)})--++(0.3,0.3*2.1333);
\draw ({0.5},{-1.0/2.1333*(0.5-1.5)})--++({atan(2.1333)-90}:0.2)--++({atan(2.1333)}:0.2)--++({atan(2.1333)+90}:0.2);

\vspace{1em}
The flow rule $\dot E_p=\gamma\dind{g}{T_p}(T_p)$ implies
\[
\dev\dot E_p=\gamma\frac{\dev T_p}{|\dev T_p|},\quad\tr\dot E_p=b|\dev\dot E_p|
\]
with $\gamma f(T_p)=0$, $f(T_p)\le0$ and $\gamma\ge0$.
\end{minipage}
\end{frame}

\begin{frame}
\frametitle{Proposal of an alternative approach}
\centering
\begin{minipage}[c]{0.53\textwidth}
\textbf{Kinematics}
\\[0.5em]
The space of configurations is $Z=\{(u,\pi,\alpha,\lambda)\}$ with the displacement $u\in H^2(\Omega,\mathbb{R}^d)$ the deviatoric plastic strain $\pi\in H^1(\Omega,\text{DevSym})$, the damage variable $\alpha\in H^1(\Omega)$ and the volumetric plastic strain $\lambda\in L^2(\Omega)$. The total plastic strain is $E_p=\pi+\frac{\lambda}{d}$ and the elastic strain is $E_e=Eu-E_p$.
\\[0.5em]
The generic rate is $w=(v,\rho,\beta,\mu)$, the plastic rate is $V_p=\rho+\frac{\mu}{d}I$ and the elastic rate is $V_e=Ev-V_p$.
\\[0.5em]
Add the kinematic constrain $z\in K=\{\lambda=L(\alpha)\}$ with $L\colon\mathbb{R}\to\mathbb{R}$
\[
\int_\Omega(\lambda-L(\alpha))qdx=0\quad\text{for all $q\in L^2(\Omega)$}.
\]
\textbf{Micropressure}
\\[0.5em]
Parametrize the constraint reaction $\mathcal{P}\in-N_K(z)$ as
\[
\dual{\mathcal{P},w}=\int_\Omega p(\mu-L'(\alpha)\beta)dx
\]
with $p\in L^2(\Omega)$ the micropressure.
\end{minipage}
\hspace{0.03\textwidth}

\begin{center}

\textbf{Notation}
\\[0.5em]
Define
\begin{gather*}
Y=\{y=(u,\pi,\alpha)\} \\
\Lambda=\{\lambda\} \\
\Lambda^*=\{p\}.
\end{gather*}
and $\mathcal{L}\colon Y\to\Lambda$ by $\mathcal{L}(y)=L(\alpha)$. Denote the virtual rate as $w=(h,\mu)$ with $h=(v,\rho,\beta)\in Y$ and $\mu\in\Lambda$. Then
\begin{gather*}
\lambda=\mathcal{L}(y) \\
\dual{\mathcal{P},w}=\dual{p,\mu}-\dual{\mathcal{L}'(y)^*p,h}.
\end{gather*}
\end{minipage}

\begin{frame}
\frametitle{Proposal of an alternative approach}
\centering
\begin{minipage}[c]{0.58\textwidth}
\textbf{Statics}
\\[0.5ex]
The internal generalized force is
\[
\dual{\mathcal{I},w}=
\begin{multlined}[t]
\int_\Omega\left[T:Ev+K\threedots\grad^2u\right]dx+\\
+\int_\Omega\left[T_e:V_e+T_p:\rho+K_p\threedots\grad\rho\right]dx+\\
+\int_\Omega\left[\tau\beta+\kappa\cdot\grad\beta\right]dx
\end{multlined}
\]
with $T\colon\Omega\to\text{Sym}$ the macroscopic stress, $K\colon\Omega\to\text{Lin}\otimes\mathbb{R}$ the macroscopic hyperstress, $T_e\colon\Omega\to\text{Sym}$ the elastic stress, $T_p\colon\Omega\to\text{DevSym}$ the plastic force, $K_p\colon\Omega\to\text{DevSym}\otimes\mathbb{R}$ the plastic stress, $\tau\colon\Omega\to\mathbb{R}$ the damage force and $\kappa\colon\Omega\to\mathbb{R}^d$ the damage stress. The external generalized force is
\[
\dual{\mathcal{X},w}=\int_\Omega f\cdot vdx+\int_{\partial\Omega}h\cdot vdx.
\]
The equilibrium is $-\mathcal{I}+\mathcal{X}+\mathcal{P}=0$.
\end{minipage}
\hspace{0.02\textwidth}
\begin{minipage}[c]{0.38\textwidth}

\begin{gather*}
\lambda_N\to\lambda\quad\text{in $C^0([0,T],\Lambda)$} \\
\bar\lambda_N\to\lambda\quad\text{in $L^2((0,T),\Lambda)$} \\
p_N\to p\quad\text{in $C^0([0,T],\Lambda^*)$} \\
\bar p_N\to p\quad\text{in $L^2((0,T),\Lambda^*)$}.
\end{gather*}
This allows to pass to the limit.\qed
