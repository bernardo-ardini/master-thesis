\begin{frame}
\frametitle{Aims and results of the thesis}
\centering
\begin{minipage}[c]{0.47\textwidth}
\textbf{Description of a mechanical system}
\\[0.5em]
The systems are the elements $z$ of a separable and reflexive Banach space $Z$.
\\[0.5em]
Generalized forces are the elements $\mathcal{F}$ of the dual space $Z^*$.
\\[1em]
\textbf{Equilibrium}
\\[0.5em]
Equilibrium is the balance
\[
-\mathcal{I}(t)+\mathcal{X}(t)=0\quad\text{for a.e.\ $t\in(0,T)$}
\]
between internal forces $\mathcal{I}\colon(0,T)\to Z^*$ and external $\mathcal{X}\colon(0,T)\to Z^*$ forces.
\end{minipage}
\hspace{0.02\textwidth}
\begin{minipage}[c]{0.47\textwidth}
\textbf{Constitutive relations}
\\[0.5em]
Let $\Psi\colon Z\to\mathbb{R}\cup\{0\}$ be the free energy and $\Phi\colon Z\to[0,\infty]$ the dissipation potential.
\\[0.5em]
Internal forces are related to the evolution $z\colon[0,T]\to Z$ with a constitutive relation
\[
\mathcal{I}(t)\in D\Psi(z(t))+\partial\Phi(\dot z(t))\quad\text{for a.e.\ $t\in(0,T)$}.
\]
\textsc{Definition.} $\xi\in\partial\Phi(v)$ if $\Phi(w)\ge\Phi(v)+\dual{\xi,w-v}$ for all $w\in Z$.
\\[0.5em]
\textsc{Proposition.} If $\Phi$ is convex and $\Phi(0)=0$ then $\dual{\xi,v}\ge0$ for all $\xi\in\partial\Phi(v)$.
\\[0.5em]
\textsc{Corollary.} $\frac{d}{dt}\Psi(z(t))\le\dual{\mathcal{I}(t),\dot z(t)}=\dual{\mathcal{X}(t),\dot z(t)}$.
\end{minipage}
\\[2em]
\begin{minipage}[c]{0.4\textwidth}
\textsc{Problem.} Given $z_0\in Z$ and $\mathcal{X}$ find $z$ such that $z(0)=z_0$ and equilibrium holds.
\end{minipage}
$\qquad\implies\qquad$
\begin{minipage}[c]{0.4\textwidth}
Doubly nonlinear equation
\[
D\Psi(z(t))+\partial\Phi(\dot z(t))\ni\mathcal{X}(t)
\]
for a.e.\ $t\in(0,T)$
\end{minipage}
\end{frame}

\begin{frame}
\frametitle{Classical approach}
\centering
\begin{minipage}[c]{0.6\textwidth}
\textbf{Kinematics}
\\[0.5em]
The space of configurations is $Z=\{(u,E_p)\}$ with the displacement $u\in H^1(\Omega,\mathbb{R}^d)$ and the plastic strain $E_p\in H^1(\Omega,\text{Sym})$. The elastic strain is $E_e=Eu-E_p$.
\\[0.5em]
The generic rate is $w=(v,V_p)$ and the elastic rate is $V_e=Ev-V_p$.
\\[1em]
\textbf{Statics}
\\[0.5ex]
The internal generalized force is
\[
\dual{\mathcal{I},w}=\int_\Omega\left[T_e:V_e+T_p:\grad V_p+K_p\threedots\grad V_p\right]dx
\]
with $T_e\colon\Omega\to\text{Sym}$ the elastic stress, $T_p\colon\Omega\to\text{Sym}$ the plastic force and $K_p\colon\Omega\to\text{Sym}\otimes\mathbb{R}$ the plastic stress. The external generalized force is
\[
\dual{\mathcal{X},w}=\int_\Omega f\cdot vdx+\int_{\partial\Omega}h\cdot vdx.
\]
\end{minipage}
\hspace{0.05\textwidth}
\begin{minipage}[c]{0.33\textwidth}
\begin{center}
\begin{tikzpicture}[scale=0.7]
\draw[fill=lime]

};

\end{tikzpicture}
\end{center}
\vspace{2em}
\textsc{Proposition.} The strong form of the equilibrium equation is
\[
\begin{gathered}
\diver T_e+f=0\\
T_e-T_e+\diver K_p=0
\end{gathered}
\quad\text{in $\Omega$}
\]
and $T_en=h$, $K_pn=0$ in $\partial\Omega$.
\end{minipage}

\begin{frame}
\frametitle{Classical approach}
\centering
\begin{minipage}[c]{0.6\textwidth}
\textbf{Constitutive relations}
\\[0.5ex]
Let the free energy
\[
\Psi(z)=\int_\Omega\left[\frac{1}{2}\mathfrak{C}E_e:E_e+\frac{1}{2}\mathfrak{b}|K_p|^2\right]dx
\]
that implies
\[
T_e=\mathfrak{C}E_e\quad\text{and}\quad K_p=\mathfrak{b}\grad E_p.
\]
\textbf{Yield condition and flow rule}
\\[0.5em]
Fix $f,g\colon\text{Sym}\to\mathbb{R}$ convex, the yield function and plastic potential respectively. Let $C=\{f\le0\}$ and $L(T_p)=\{g\le g(T_p)\}$.
\\[0.5em]
It is required that $T_p\in C$ with $C\subset\text{Sym}$ closed, convex and with $0\in\inter C$. The yield condition is $T_p\in\partial C$ and the flow rule is $\dot E_p=0$ if $T_p\in\inter C$
\[
\dot E_p\in
\begin{cases}
\{0\} & \text{if $T_p\in\inter C$} \\
N_{L(T_p)}(T_p) & \text{if $T_p\in\partial C$}.
\end{cases}
\]

\begin{frame}
\frametitle{Internal virtual power}

Let $\tau$ macroscopic Kirchhoff stress and $\tau_p$ microstress
\[
\tilde{\mathcal{W}}=\int_E\left(\tau\;\tilde{\dot\epsilon}_e+\tau_p\;\tilde{\dot\epsilon}_p \right)Vde,
\]

\begin{lem}
We can write $\tilde{\mathcal{W}}$ as
\[
\tilde{\mathcal{W}}=-\sum_{a\in A}f_a\cdot\tilde{\dot x}_a+\int_E(\tau_p-\tau)\;\tilde{\dot\epsilon}_pVde
\]
with
\[
f_a=\int_Es_aqnde\qquad s_a(e)=
\begin{cases}
1 & \text{if $a=e_1$} \\
-1 & \text{if $a=e_2$} \\
0 & \text{otherwise}
\end{cases}
\]
\end{lem}
\begin{proof}
\[
\tau\tilde{\dot\epsilon}V=\frac{v}{V}\sigma\frac{\tilde{\dot l}}{l}V=a\sigma n\cdot\left(\tilde{\dot x}_b-\tilde{\dot x}_a\right)
\]
\end{proof}
\end{frame}

\begin{frame}
\frametitle{Dirichlet, Neumann and contact conditions}
\centering
We split $A=A_D\cup A_N\cup A_C$

\begin{columns}
\begin{column}{0.3\framewidth}
\begin{block}{Dirichlet condition}
If $a\in A_D$ we impose $x_a=\overline{x}_a$ and call $r_a\in\mathbb{R}^d$ the Lagrange multiplier
\end{block}
\end{column}
\begin{column}{0.3\framewidth}
\begin{block}{Neumann condition}
On $a\in A_N$ acts the external load $h_a\in\mathbb{R}^d$
\end{block}

\end{column}
\end{columns}

\begin{columns}
\begin{column}{0.6\framewidth}
\begin{block}{Contact condition}
If $a\in A_C$ let $\gamma_a=\phi(x_a,t)$ with
$\omega^M=\left\{x\in\mathbb{R}^d\middle|\phi(x,t)\le0\right\}$ the obstacle and $\kappa_a\in\mathbb{R}$ the Lagrange multiplier
\[
\gamma_a\ge0\quad\kappa_a\ge0\quad\gamma_a\kappa_a=0
\]
\end{block}
\end{column}
\end{columns}

\end{frame}

\begin{frame}
\frametitle{External virtual power and balances}

\begin{columns}
\begin{column}{0.45\framewidth}
\[
\tilde{\mathcal{P}}=\sum_{a\in A_D}r_a\tilde{v}_a+\sum_{a\in A_N}h_a\tilde{v}_a+\sum_{a\in A_C}\kappa_a\tilde{\dot\gamma}_a.
\]
\centering notice that
\[
\tilde{\dot\gamma}_a=\nabla\phi(x_a,t)\tilde{v}_a
\]
\end{column}
\begin{column}{0.4\framewidth}
\centering recall
\[
\tilde{\mathcal{W}}=-\sum_{a\in A}f_a\cdot\tilde{\dot x}_a+\int_E(\tau_p-\tau)\;\tilde{\dot\epsilon}_pVde
\]
\end{column}
\end{columns}

\vfill

\begin{columns}
\begin{column}{0.4\framewidth}
\begin{defi}[Virtual Powers Principle]
equilibrium holds when $\tilde{\mathcal{W}}=\tilde{\mathcal{P}}$ for all virtual rates
\end{defi}
\end{column}
\begin{column}{0.45\framewidth}
\begin{block}{Equilibrium equations}
in $A_D$ we have $f_a+r_a=0$ \\
in $A_N$ we have $f_a+h_a=0$ \\
in $A_C$ we have $f_a+\kappa_a\nabla\phi(x_a,t)=0$
on each truss $\tau_p-\tau=0$
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{Initial boundary value problem}
\begin{prb}
Let $\overline{x}_a(t)$, $h_a(t)$ and $\phi(x,t)$ be given. Find $x_a(t)$ with $x_a(0)=X_a$, $r(t)$, $\kappa(t)$, $\epsilon_p(e,t)$ with $\epsilon_p(e,0)=0$ and $\alpha(e,t)$ with $\alpha(e,0)=0$ such that for all $t\in[0,T]$
\begin{align*}
&x_a(t)=\overline{x}_a(t)\quad\text{and}\quad f_a+r_a=0\quad\text{for $a\in A_D$} \\
&f_a+h_a=0\quad\text{for $a\in A_N$} \\
&f_a+\kappa_a\nabla\phi(x_a,t)=f_a+c_a=0\quad\text{for $a\in A_C$} \\
&\gamma_a\ge0,\quad \kappa_a\ge 0\quad\text{and}\quad\gamma_a\kappa_a=0\quad\text{for $a\in A_C$} \\
&\tau_p=\tau\quad\text{in $E$} \\
&\dot\alpha=r_p\quad\text{in $E$}.
\end{align*}
\end{prb}
\end{frame}

\begin{frame}
\frametitle{Numerical approximation}
\begin{prb*}
Let $\overline{\boldsymbol{x}}(t)$, $\boldsymbol{h}(t)$ and $\phi(x,t)$ be given. Find $\boldsymbol{x}(t)$ with $\boldsymbol{x}(0)=\boldsymbol{X}$, $\boldsymbol{r}(t)$, $\boldsymbol{\kappa}(t)$, $\boldsymbol{\epsilon}_p(t)$ with $\boldsymbol{\epsilon}_p(0)=0$ and $\boldsymbol{\alpha}(t)$ with $\boldsymbol{\alpha}(0)=0$ such that for all $t$
\begin{align*}
& \boldsymbol{f}+\boldsymbol{H}_D\boldsymbol{r}+\boldsymbol{H}_N\boldsymbol{h}+\boldsymbol{H}_C\boldsymbol{c}=0 \\
& \boldsymbol{H}_D^T\boldsymbol{x}=\overline{\boldsymbol{x}} \\
\label{eqn:truss-KT}
& \boldsymbol\gamma_a\ge0,\quad\boldsymbol\kappa_a\ge 0\quad\text{and}\quad\boldsymbol\gamma_a\boldsymbol\kappa_a=0\quad\text{(KT)} \\
& \boldsymbol\tau_p=\boldsymbol\tau \\
& \dot{\boldsymbol\alpha}=\boldsymbol r_p.
\end{align*}
\end{prb*}

\begin{columns}
\begin{column}{0.4\textwidth}
\begin{block}{\centering Fischer--Burmeister function}
\centering we write the condition (KT) as $\boldsymbol{\Phi}(\boldsymbol{\gamma},\boldsymbol{\kappa})=0$ where
\[
\boldsymbol\Phi_a(\boldsymbol\gamma,\boldsymbol\kappa)=\boldsymbol\gamma_a+c\boldsymbol\kappa_a-|\boldsymbol\gamma_a-c\boldsymbol\kappa_a|
\]
\end{block}
\end{column}

\begin{column}{0.4\textwidth}
\begin{block}{\centering Residual}
we write the five sets of equalities as $\boldsymbol{R}=0$ where $\boldsymbol{R}$ depends on $\boldsymbol{x}$, $\boldsymbol{r}$, $\boldsymbol{\kappa}$, $\boldsymbol{\epsilon}_p$, $\boldsymbol{\eta}_p$, $\boldsymbol{\alpha}$, $\boldsymbol{\dot\alpha}$ and $t$
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{Time discretization}

\begin{columns}
\begin{column}{0.4\textwidth}
At time $t-\Delta t$ we know the solution and we look for the solution at time $t$. We set
\begin{gather*}
\boldsymbol{x}=\boldsymbol{x}^-+\Delta t\;\boldsymbol{v} \\
\boldsymbol{\epsilon}_p=\boldsymbol{\epsilon}_p^-+\Delta t\;\boldsymbol{\eta}_p \\
\boldsymbol{\alpha}=\boldsymbol{\alpha}^-+\Delta t\;\dot{\boldsymbol{\alpha}}.
\end{gather*}
Now $\boldsymbol{R}$ depends on $\boldsymbol{v}$, $\boldsymbol{r}$, $\boldsymbol{\kappa}$, $\boldsymbol{\eta}_p$ and $\dot{\boldsymbol{\alpha}}$ that we collect in $\boldsymbol{\xi}$.
\end{column}
\begin{column}{0.4\textwidth}
\begin{block}{Newton method}
We solve
\[
\boldsymbol{R}(\boldsymbol{\xi})=0
\]
with Newton method
\begin{itemize}
\item[-] set $\boldsymbol\xi_0=\boldsymbol\xi^-$
\item[-] impose $\boldsymbol{R}(\xi_{i-1}+\boldsymbol{\delta}_i)\approxeq\boldsymbol{R}(\xi_{i-1})+\boldsymbol{K}(\xi_{i-1})\delta_i=0$
\item[-] solve $\boldsymbol{\delta}_i=-\boldsymbol{K}(\xi_{i-1})^{-1}\boldsymbol{R}(\xi_{i-1})$
\item[-] update $\boldsymbol{\xi}_i=\boldsymbol{\xi}_{i-1}+\eta_i\boldsymbol\delta_i$
\end{itemize}
\end{block}
\end{column}
\end{columns}

\begin{columns}
\begin{column}{0.5\framewidth}
\begin{block}{Time increment}
Choose $\Delta t_0$, $\Delta t_\text{max}$, $\Delta t_\text{min}$, $k_\text{max}$ and $k_\text{min}$
\[
\Delta t_k=
\begin{cases}
\Delta t_0\left[\frac{\Delta t_\text{max}}{\Delta t_0}\right]^\frac{k}{k_\text{max}}\quad & \text{if $k\ge0$} \\
\Delta t_0\left[\frac{\Delta t_0}{\Delta t_\text{min}}\right]^\frac{k}{k_\text{min}} & \text{if $k\le0$}
\end{cases}
\]
Start with $k=0$ and decrement it if the step fails
\end{block}
\end{column}
\begin{column}{0.3\framewidth}
\begin{block}{\centering Failure criteria}
\vspace{-2em}
\begin{align*}
& \text{ITER}>\text{MAXITER} \\
&|\boldsymbol{x}-\boldsymbol{x}^*|> \text{TOL}(\boldsymbol{x}) \\
&|\boldsymbol{r}-\boldsymbol{r}^-|> \text{TOL}(\boldsymbol{r}) \\
&|\boldsymbol{\kappa}-\boldsymbol{\kappa}^-|> \text{TOL}(\boldsymbol{\kappa}) \\
&|\boldsymbol{\epsilon}_p-\boldsymbol{\epsilon}_p^*|> \text{TOL}(\boldsymbol{\epsilon}_p) \\
&|\boldsymbol{\alpha}-\boldsymbol{\alpha}^*|> \text{TOL}(\boldsymbol{\alpha}).
\end{align*}
with $x^*=x^-+\Delta t\;v^-$ etc.
\end{block}
\end{column}
\end{columns}

\end{frame}

\begin{columns}
\begin{column}{0.35\textwidth}
\begin{block}{Bodies and deformations}

we decompose
\[
\partial\Omega^\alpha=\Gamma_D^\alpha\cup\Gamma_N^\alpha\cup\Gamma_C^\alpha
\]
\end{block}
\begin{block}{Dirichlet condition}
on $\Gamma_D^\alpha$ we impose
\[
\chi^\alpha=\overline{\chi}^\alpha
\]
let $r^\alpha\colon\Gamma_D\to\mathbb{R}^d$ the corresponding Lagrange multiplier
\end{block}

\begin{block}{Projection}
defined as $\Pi\colon\Gamma_C^S\times[0,T]\to\Gamma_C^M$ such that

\Pi(X^S,t)\in\argmin_{X^M\in\Gamma_C^M}|\chi^S(X_S,t)-\chi^M(X_M,t)|
\]
\end{block}
\begin{block}{Gap vector and gap distance}
the gap vector is
\[
g(X^S,t)=\chi^S(X^S,t)-\chi^M(\Pi(X^S,t),t)

and the gap distance is
\[
\gamma(X^S,t)=g(X^S,t)\cdot n^M(\chi^M(\Pi(X^S,t),t),t)

notice that $g=\gamma n^M$
\end{block}

let $\kappa\colon\Gamma_C^S\to\mathbb{R}$ the Lagrange multiplier
\[
\gamma\ge0\quad\kappa\ge0\quad\gamma\kappa=0

\begin{frame}
\frametitle{Function spaces}

\centering
space of deformation
\[
\mathscr{X}^\alpha=W^{1,p}(\Omega^\alpha)^d
\]
space of virtual velocities
\[
\mathscr{V}^\alpha=W^{1,p}(\Omega^\alpha)^d
\]
space of Dirichlet reactions
\[
\mathcal{R}^\alpha=\left[W^{1-\frac{1}{p},p}(\Gamma_D^\alpha)^d\right]^*
\]
cone of admissible gap distances
\[
\mathscr{G}=W^{1-\frac{1}{p},p}(\Gamma_C^S)\qquad\mathscr{C}=\left\{\gamma\in\mathscr{G}\middle|\gamma\ge0\right\}
\]
cone of admissible contact reactions
\[
\mathscr{K}=\mathscr{G}^*\qquad\mathscr{D}=\left\{\kappa\in\mathscr{K}\middle|\text{$\dual{\kappa,\tilde\gamma}\ge0$ for all $\tilde\gamma\in\mathscr{C}$}\right\}
\]
\end{frame}

\begin{frame}
\frametitle{Weak form of the contact condition}
\begin{lem}
Let $\gamma\in\mathscr{C}$. Then $\kappa\in N_\mathscr{C}(\gamma)$ if and only if $\kappa=0$ in $\Gamma_I^S(\gamma)$ and $\kappa\ge0$ in $\Gamma_A^S(\gamma)$.
\end{lem}
\begin{proof}
We have $\kappa\in N_\mathscr{C}(\gamma)$ if and only if
\[
\int_{\Gamma_C^S}\kappa(\tilde\gamma-\gamma)dA=\int_{\Gamma_A^S(\gamma)}\kappa\tilde\gamma dA+\int_{\Gamma_I^S(\gamma)}\kappa(\tilde\gamma-\gamma)dA\ge0
\]
for all $\tilde\gamma\in\mathscr{C}$. Take $\tilde\gamma\ge0$ arbitrary with support in $\Gamma_A^S(\gamma)$ so $\kappa\ge0$ there. On $\Gamma_I^S(\gamma)$ we have $\gamma>0$. Take $\tilde\gamma\ge0$ arbitrary with support in $\Gamma_I^S(\gamma)$ so $\kappa=0$ there.
\end{proof}
\end{frame}

\begin{frame}
\frametitle{Weak form of the contact condition}
\begin{lem}
The condition $\gamma\in\mathscr{C}$ and $\kappa\in N_\mathscr{C}(\gamma)$ is equivalent to $\kappa\in\mathscr{D}$ and $\gamma\in N_\mathscr{D}(\kappa)$.
\end{lem}
\begin{proof}
Using a theorem form Convex Analysis the condition $\gamma\in\mathscr{C}$ and $\kappa\in N_\mathscr{C}(\gamma)$ is equivalent to $-\kappa\in\dom\sigma_\mathscr{C}$ and $\gamma\in\partial^s\sigma_\mathscr{C}(-\kappa)$. Here $\sigma_\mathscr{C}=\iota_\mathscr{C}^*$ is the support function of $\mathscr{C}$. We have
\[
\sigma_\mathscr{C}(-\kappa)=\sup_{\tilde\gamma\in\mathscr{C}}(-\dual{\kappa,\tilde\gamma}).
\]
Since $\tilde\gamma\in\mathscr{C}$ implies $t\tilde\gamma\in\mathscr{C}$ for all $t\ge0$ then $-\kappa\in\dom\sigma_\mathscr{C}$ if and only if $\kappa\in\mathscr{D}$. Moreover $\sigma_\mathscr{C}$ is the indicator function of $-\mathscr{D}$. So $\gamma\in\partial^s\sigma_\mathscr{C}(-\kappa)$ if and only if $\gamma\in N_\mathscr{D}(\kappa)$.
\end{proof}
\begin{rmk*}
Formally $\gamma\in N_\mathscr{D}(\kappa)$ means
\[
\int_{\Gamma_C^S}(\tilde\kappa-\kappa)\gamma dA\ge0
\]
for all $\tilde\kappa\in\mathscr{D}$
\end{rmk*}
\end{frame}

\begin{frame}
\frametitle{Internal virtual power}
\[
\tilde{\mathscr{\mathcal{W}}}=\sum_\alpha\int_{\Omega^\alpha} P^\alpha:\Grad\tilde v^\alpha dX-\int_{\Gamma_C^S}\kappa\tilde{\dot\gamma}dA.
\]
can be rewritten as
\[
\begin{split}
\tilde{\mathcal{W}}=&\sum_\alpha\bigg[-\int_{\Omega^\alpha}\diver(P^\alpha)\cdot\tilde v^\alpha dX+\int_{\Gamma_D^\alpha}P^\alpha N^\alpha\cdot\tilde v^\alpha dA+\int_{\Gamma_N^\alpha}P^\alpha N^\alpha\cdot\tilde v^\alpha dA+\\
&+\int_{\Gamma_I^\alpha}P^\alpha N^\alpha\cdot\tilde v^\alpha dA\bigg]+ \\
&+\int_{\Gamma_A^S}\left(P^S N^S+\kappa\;n^S(\chi^S,t)\right)\cdot\tilde v^SdA+\\
&+\int_{\Gamma_A^M}\left(P^M N^M+C_M^S\;\kappa(\Pi^{-1},t)\;n^M(\chi^M,t)\right)\cdot\tilde v^MdA
\end{split}
\]
\end{frame}

\begin{frame}
\frametitle{External virtual power and balances}
\[
\tilde{\mathcal{P}}=\sum_\alpha\left[\int_{\Omega^\alpha}f^\alpha\cdot\tilde v^\alpha dX+\int_{\Gamma_D^\alpha}r^\alpha\cdot\tilde v^\alpha dA+\int_{\Gamma_N^\alpha}h^\alpha\cdot\tilde v^\alpha dA\right].
\]
we get the balances
\begin{align*}
&\diver(P^\alpha)+f^\alpha=0\quad\text{on $\Omega^\alpha$} \\
&P^\alpha N^\alpha=r^\alpha\quad\text{on $\Gamma_D^\alpha$} \\
&P^\alpha N^\alpha=h^\alpha\quad\text{on $\Gamma_N^\alpha$} \\
&P^\alpha N^\alpha=0\quad\text{on $\Gamma_I^\alpha$} \\
&N^\alpha=-\kappa\;n^S(\chi^S,t)\quad\text{on $\Gamma_A^S$} \\
&P^MN^M=-C^S_M\;\kappa(\Pi^{-1},t)\;n^M(\chi^M,t)\quad\text{on $\Gamma_A^M$}
\end{align*}
\end{frame}

\begin{frame}
\frametitle{Initial boundary value problem}
\begin{prb}
Let $\overline{\chi}^\alpha(X^\alpha,t)$ with $\overline{\chi}^\alpha(X^\alpha,0)=X^\alpha$, $f^\alpha(X^\alpha,t)$ and $h^\alpha(X^\alpha,t)$ be given. For all $t\in[0,T]$ find $\chi^\alpha(t)\in\mathscr{X}^\alpha$ with $\chi^\alpha(X^\alpha,0)=X^\alpha$, $r^\alpha(t)\in\mathscr{R}^\alpha$ and $\kappa(t)\in\mathscr{D}$ such that
\begin{align*}
&\tilde{\mathcal{W}}=\tilde{\mathcal{P}}\quad\text{for all $\tilde v^\alpha\in\mathscr{V}^\alpha$} \\
&\int_{\Gamma_D^\alpha}(\chi^\alpha-\overline{\chi}^\alpha)\cdot\tilde{r}^\alpha dA=0\quad\text{for all $\tilde{r}^\alpha\in\mathscr{R}^\alpha$} \\
&\int_{\Gamma_C^S}(\tilde\kappa-\kappa)\gamma dA\ge0\quad\text{for all $\tilde\kappa\in\mathscr{D}$}.
\end{align*}
\end{prb}
\end{frame}

\begin{frame}
\frametitle{Numerical approximation}
We take the deformation of the form
\[
\chi(X)=N(X)\boldsymbol{\chi}\qquad\boldsymbol{\chi}\in\mathbb{R}^N\qquad N\colon\Omega\to\mathbb{R}^{d\times N}
\]
and the virtual velocity
\[
\tilde{v}(X)=N(X)\tilde{\boldsymbol{v}}\qquad\tilde{\boldsymbol{v}}\in\mathbb{R}^N
\]
Moreover
\[
r(X)=L(X)\boldsymbol{r}\qquad\kappa(X)=M(X)\boldsymbol{\kappa}
\]
with
\[\boldsymbol{r}\in\mathbb{R}^L\qquad\boldsymbol{\kappa}\in\mathbb{R}^M
\]
and
\[
L\colon\Gamma_D\to\mathbb{R}^{d\times L}\qquad M\colon\Gamma_C\to\mathbb{R}^{1\times M}
\]
We redefine the function spaces accordingly and we set
\[
\mathscr{D}=\left\{\kappa\in\mathscr{K}\middle|\text{$\boldsymbol{\kappa}_a\ge0$ for $a\in\{1,\dots,M\}$}\right\}.
\]
\end{frame}

\begin{frame}
\frametitle{Residual}
\[
\int_\Omega\hat{P}\left(\dind{N}{X}\boldsymbol\chi\right):\dind{N}{X}\tilde{\boldsymbol{v}}dX-\int_{\Gamma_C}M\boldsymbol\kappa\; G\tilde{\boldsymbol{v}}dA=\int_\Omega f\cdot N\tilde{\boldsymbol{v}}dX+\int_{\Gamma_D}L\boldsymbol r\cdot N\tilde{\boldsymbol{v}}dA+\int_{\Gamma_N}h\cdot N\tilde{\boldsymbol{v}}dA
\]
\centering gives
\begin{multline*}
\boldsymbol{R}_1=\int_\Omega\left(\dind{N}{X}\right)^T:\hat{P}\left(\dind{N}{X}\boldsymbol\chi\right)dX-\int_{\Gamma_C}G^T\;M\boldsymbol\kappa dA-\\
-\int_\Omega N^TfdX-\int_{\Gamma_D}N^TL\boldsymbol rdA-\int_{\Gamma_N}N^ThdA=0.
\end{multline*}
\end{frame}

\begin{frame}
\frametitle{Residual}
\[
\int_{\Gamma_D}(N\boldsymbol{\chi}-\overline{\chi})\cdot L\tilde{\boldsymbol{r}}dA=0
\]
\centering gives
\[
\boldsymbol{R}_2=\int_{\Gamma_D}L^T(N\boldsymbol{\chi}-\overline{\chi})dA=0
\]
\end{frame}

\begin{frame}
\frametitle{Residual}
\[
0\le\int_{\Gamma_C}(\tilde\kappa-\kappa)\gamma dA=\boldsymbol{\gamma}^T(\tilde{\boldsymbol{\kappa}}-\boldsymbol{\kappa})\qquad\boldsymbol\gamma=\int_{\Gamma_C^S}M^T\gamma dA
\]
\centering is equivalent to
\[
\boldsymbol{\gamma}_a\ge0\qquad\boldsymbol{\kappa}_a\ge0\qquad\boldsymbol{\gamma}_a\boldsymbol{\kappa}_a=0
\]
which we write as
\[
\boldsymbol{R}_3=\Phi(\boldsymbol{\gamma},\boldsymbol{\kappa})=0
\]
\end{frame}

\begin{frame}

\[
\boldsymbol{R}=
\begin{pmatrix}
\boldsymbol{R}_1 \\
\boldsymbol{R}_2 \\

\end{pmatrix}
\qquad
\boldsymbol{K}=
\begin{pmatrix}

\boldsymbol{K}_{21} & 0 & 0 \\

\end{pmatrix}.
\]
We have that
\begin{align*}
\boldsymbol{K}_{11}&=\Delta t\int_\Omega\left(\dind{N}{X}\right)^T:\hat{P}\dind{N}{X}dX+\Delta t\int_\Omega\left(\dind{N}{X}\right)^T:\left(\dind{\hat{P}}{F}\dind{N}{X}\right)\left(\dind{N}{X}\boldsymbol\chi\right)dX \\

\boldsymbol{K}_{31}=\dind{\boldsymbol{\Phi}}{\boldsymbol{\gamma}}\dind{\boldsymbol{\gamma}}{\boldsymbol{v}},\qquad\boldsymbol{K}_{33}=\dind{\boldsymbol{\Phi}}{\boldsymbol{\kappa}}.
\end{gather*}
We notice that
\[
\dind{\boldsymbol{\gamma}}{\boldsymbol{v}}=\Delta t\;\dind{\boldsymbol{\Phi}}{\boldsymbol{\gamma}}\int_{\Gamma_C}M^TGdA.
\]
\end{frame}
